CREATE PROCEDURE 'userAddOrEdit'(
IN _ID INT,
IN _eID varchar(10),
IN _password varchar(20),
IN _name varchar(50),
IN _email varchar(50),
IN _phone varchar(15),
IN _year int
)
BEGIN
	IF _ID = 0 THEN
		INSERT INTO user(eID,password,name,email,phone,year)
		VALUES(_eID,_password,_name,_email,_phone,_year);
		SET ID = LAST_INSERT_ID();
	ELSE
		UPDATE user
		SET
		eID = _eID,
		password = _password,
		name = _name,
		email = _email,
		phone = _phone,
		year = _year
		WHERE ID = _ID;
	END IF;

	SELECT ID AS 'ID';
END
